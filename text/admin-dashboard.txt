<script>
    const toggleSidebar = () => {
        document.getElementById('sidebar').classList.toggle('show');
    }

    const showTab = (tabId) => {
        document.querySelectorAll('.tab-content').forEach(tab => tab.classList.add('d-none'));
        document.getElementById(tabId).classList.remove('d-none');
        document.querySelectorAll('.sidebar a').forEach(link => link.classList.remove('active'));
        document.querySelector(`.sidebar a[href='#${tabId}']`).classList.add('active');
        if (window.innerWidth < 768) toggleSidebar();
    }

    const submitForm = (type) => {
        let activeTab = '';
        if(type === 'Movie') activeTab = 'addmovie';
        else if(type === 'Webseries') activeTab = 'addwebseries';
        else if(type === 'Podcasts') activeTab = 'addpodcast';
        else if(type === 'Kids') activeTab = 'kids';

        if(activeTab) {
            const inputs = document.querySelectorAll(`#${activeTab} input, #${activeTab} select`);
            
            let isValid = true;

            inputs.forEach(input => {
                if(input.value.trim() === "" && input.type !== 'file'){
                    isValid = false;
                    input.style.border = "2px solid red"; // highlight empty input
                } else {
                    input.style.border = ""; // reset border if filled
                }
            });

            if(!isValid){
                alert("Please fill all the fields before submitting!");
                return; 
            }

            alert(type + " added successfully!");

            // Clear inputs
            inputs.forEach(input => {
                if(input.type === 'file') {
                    input.value = null;
                } else {
                    input.value = '';
                }
            });
        }
    }
</script>
